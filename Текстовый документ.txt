            }

            return {
                id: "guest_local",
                first_name: "MAX",
                last_name: "Клиент",
                username: "guest",
                photo_url: "",
                language_code: "ru"
            };
        }

        function getUserDisplayName(user) {
            return [user.first_name, user.last_name].filter(Boolean).join(" ").trim() || "MAX Клиент";
        }

        function getUserTag(user) {
            return user.username ? "@" + user.username : "id" + user.id;
        }

        function getUserStorageKey(userId) {
            return `max_orders_user_${userId}`;
        }

        function saveUserProfile(user) {
            const users = JSON.parse(localStorage.getItem(STORAGE_USERS_KEY)) || {};
            users[user.id] = {
                id: user.id,
                first_name: user.first_name,
                last_name: user.last_name,
                username: user.username,
                photo_url: user.photo_url,
                language_code: user.language_code,
                last_seen_at: new Date().toISOString()
            };
            localStorage.setItem(STORAGE_USERS_KEY, JSON.stringify(users));
        }

        function getAllOrders() {
            return JSON.parse(localStorage.getItem(STORAGE_ORDERS_KEY)) || [];
        }

        function setAllOrders(orders) {
            localStorage.setItem(STORAGE_ORDERS_KEY, JSON.stringify(orders));
        }

        function getUserOrders(userId) {
            return JSON.parse(localStorage.getItem(getUserStorageKey(userId))) || [];
        }

        function setUserOrders(userId, orders) {
            localStorage.setItem(getUserStorageKey(userId), JSON.stringify(orders));
        }

        function generateOrderId() {
            const part1 = Date.now().toString(36).toUpperCase();
            const part2 = Math.random().toString(36).slice(2, 6).toUpperCase();
            return `MAX-${part1}-${part2}`;
        }

        function formatDate(dateString) {
            const d = new Date(dateString);
            if (isNaN(d.getTime())) return dateString;
            return d.toLocaleString('ru-RU', {
                day: '2-digit',
                month: '2-digit',
                year: 'numeric',
                hour: '2-digit',
                minute: '2-digit'
            });
        }

        function escapeHTML(str) {
            return String(str ?? '')
                .replace(/&/g, '&amp;')
                .replace(/</g, '&lt;')
                .replace(/>/g, '&gt;')
                .replace(/"/g, '&quot;')
                .replace(/'/g, '&#039;');
        }

        function escapeCSV(value) {
            const str = String(value ?? '');
            return `"${str.replace(/"/g, '""')}"`;
        }

        function renderProfile() {
            const user = currentUser;
            if (!user) return;

            const userNameEl = document.getElementById('user-name');
            const userIdEl = document.getElementById('user-id');
            const avatarEl = document.getElementById('avatar');
            const statusEl = document.getElementById('user-status');
            const extraEl = document.getElementById('user-extra');

            userNameEl.innerText = getUserDisplayName(user);
            userIdEl.innerText = getUserTag(user);

            if (user.photo_url) {
                avatarEl.innerHTML = `<img src="${escapeHTML(user.photo_url)}" alt="avatar">`;
            } else {
                avatarEl.textContent = (user.first_name || "?").charAt(0).toUpperCase();
            }

            statusEl.innerText = isAdminUser() ? "MAX Admin Access" : "MAX Premium Guest";
            extraEl.innerText = `Профиль подключен к Mini App. ID: ${user.id}`;

            document.getElementById('admin-panel').style.display = isAdminUser() ? 'block' : 'none';
        }

        function renderCats() {
            const cats = ["Все", ...new Set(PRODUCTS.map(p => p.cat))];
            document.getElementById('cat-nav').innerHTML = cats.map(c => `
                <div class="cat-btn ${c === currentCat ? 'active' : ''}" onclick="setCat('${String(c).replace(/'/g, "\\'")}')">${c}</div>
            `).join('');
        }

        function setCat(c) {
            currentCat = c;
            renderCats();
            renderProducts();
        }

        function renderProducts() {
            const grid = document.getElementById('product-grid');
            const filtered = currentCat === "Все" ? PRODUCTS : PRODUCTS.filter(p => p.cat === currentCat);

            grid.innerHTML = filtered.map(p => `
                <div class="card">
                    <img src="${escapeHTML(p.img)}" class="p-img" loading="lazy" alt="${escapeHTML(p.name)}">
                    <div class="p-name serif">${escapeHTML(p.name)}</div>
                    <div class="p-price">${p.price} ₽</div>
                    <button class="add-btn" onclick="addToCart(${p.id})">В корзину</button>
                </div>
            `).join('');
        }

        function addToCart(id) {
            const p = PRODUCTS.find(x => x.id === id);
            if (!p) return;

            const item = cart.find(x => x.id === id);
            if (item) item.qty++;
            else cart.push({ ...p, qty: 1 });

            safeHaptic('medium');
            updateCartUI();
        }

        function changeQty(id, delta) {
            const item = cart.find(x => x.id === id);
            if (!item) return;

            item.qty += delta;
            if (item.qty <= 0) {
                cart = cart.filter(x => x.id !== id);
            }

            updateCartUI();
        }

        function updateCartUI() {
            localStorage.setItem(STORAGE_CART_KEY, JSON.stringify(cart));

            const list = document.getElementById('cart-list');
            const total = cart.reduce((s, i) => s + (i.price * i.qty), 0);
            const count = cart.reduce((s, i) => s + i.qty, 0);

            const badge = document.getElementById('cart-badge');
            badge.innerText = count;
            badge.style.display = count > 0 ? 'block' : 'none';

            if (cart.length > 0) {
                document.getElementById('cart-empty').style.display = 'none';
                document.getElementById('checkout-block').style.display = 'block';

                list.innerHTML = cart.map(i => `
                    <div class="cart-item">
                        <img src="${escapeHTML(i.img)}" class="cart-img" alt="${escapeHTML(i.name)}">
                        <div class="cart-info" style="flex:1;">
                            <h4 class="serif" style="margin:0 0 6px;">${escapeHTML(i.name)}</h4>
                            <div style="font-weight:600;">${i.price} ₽</div>
                        </div>
                        <div class="cart-ctrl">
                            <button class="qty-btn" onclick="changeQty(${i.id}, -1)">-</button>
                            <span style="font-weight:600;">${i.qty}</span>
                            <button class="qty-btn" onclick="changeQty(${i.id}, 1)">+</button>
                        </div>
                    </div>
                `).join('');

                if (tg && tg.MainButton) {
                    try {
                        tg.MainButton.setText(`ПОДТВЕРДИТЬ ЗАКАЗ (${total} ₽)`);
                        tg.MainButton.show();
                        if (tg.MainButton.offClick) tg.MainButton.offClick(finalizeOrder);
                        tg.MainButton.onClick(finalizeOrder);
                    } catch (e) {}
                }
            } else {
                document.getElementById('cart-empty').style.display = 'block';
                document.getElementById('checkout-block').style.display = 'none';
                list.innerHTML = '';

                if (tg && tg.MainButton) {
                    try {
                        if (tg.MainButton.offClick) tg.MainButton.offClick(finalizeOrder);
                        tg.MainButton.hide();
                    } catch (e) {}
                }
            }
        }

        function saveOrderLocally(order) {
            const userOrders = getUserOrders(order.user.id);
            userOrders.unshift(order);
            setUserOrders(order.user.id, userOrders);

            const allOrders = getAllOrders();
            allOrders.unshift(order);
            setAllOrders(allOrders);
        }

        function buildTelegramMessage(order) {
            const itemsStr = order.items
                .map(i => `• ${i.name} (${i.qty} шт) - ${i.price * i.qty}₽`)
                .join('\n');

            const userLine = `${order.user.first_name}${order.user.last_name ? ' ' + order.user.last_name : ''}`;
            const userTag = order.user.username ? '@' + order.user.username : 'id' + order.user.id;

            return [
                `🚢 *НОВЫЙ ЗАКАЗ MAX*`,
                ``,
                `🆔 Заказ: *${order.order_id}*`,
                `👤 Клиент: ${userLine} (${userTag})`,
                `📞 Тел: ${order.phone}`,
                `📍 Адрес: ${order.address}`,
                `🕒 Дата: ${formatDate(order.created_at)}`,
                ``,
                `🛒 *Состав заказа:*`,
                `${itemsStr}`,
                ``,
                `💰 *ИТОГО: ${order.total} ₽*`
            ].join('\n');
        }

        async function sendMessageToAdmin(order) {
            const message = buildTelegramMessage(order);

            const url = `https://api.telegram.org/bot${MAX_TOKEN}/sendMessage`;
            const payload = {
                chat_id: ADMIN_ID,
                text: message,
                parse_mode: 'Markdown'
            };

            const res = await fetch(url, {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify(payload)
            });

            return res.ok;
        }

        function buildOrderObject(phone, address) {
            const total = cart.reduce((s, i) => s + (i.price * i.qty), 0);

            return {
                order_id: generateOrderId(),
                created_at: new Date().toISOString(),
                phone: phone.trim(),
                address: address.trim(),
                total,
                status: 'sent_to_admin',
                status_text: 'Отправлен',
                user: {
                    id: currentUser.id,
                    first_name: currentUser.first_name,
                    last_name: currentUser.last_name,
                    username: currentUser.username,
                    photo_url: currentUser.photo_url
                },
                items: cart.map(i => ({
                    id: i.id,
                    cat: i.cat,
                    name: i.name,
                    price: i.price,
                    qty: i.qty,
                    img: i.img
                }))
            };
        }

        function buildOrdersCSV() {
            const orders = getAllOrders();

            const header = [
                'order_id',
                'created_at',
                'user_id',
                'first_name',
                'last_name',
                'username',
                'phone',
                'address',
                'item_id',
                'category',
                'item_name',
                'price',
                'qty',
                'line_total',
                'order_total',
                'status'
            ];

            const rows = [header.join(',')];

            orders.forEach(order => {
                order.items.forEach(item => {
                    rows.push([
                        escapeCSV(order.order_id),
                        escapeCSV(order.created_at),
                        escapeCSV(order.user.id),
                        escapeCSV(order.user.first_name),
                        escapeCSV(order.user.last_name),
                        escapeCSV(order.user.username),
                        escapeCSV(order.phone),
                        escapeCSV(order.address),
                        escapeCSV(item.id),
                        escapeCSV(item.cat),
                        escapeCSV(item.name),
                        escapeCSV(item.price),
                        escapeCSV(item.qty),
                        escapeCSV(item.price * item.qty),
                        escapeCSV(order.total),
                        escapeCSV(order.status_text || order.status)
                    ].join(','));
                });
            });

            return rows.join('\n');
        }

        async function sendCSVToAdmin(blob, filename) {
            const formData = new FormData();
            formData.append('chat_id', ADMIN_ID);
            formData.append('caption', `CSV выгрузка заказов MAX от ${formatDate(new Date().toISOString())}`);
            formData.append('document', blob, filename);

            const url = `https://api.telegram.org/bot${MAX_TOKEN}/sendDocument`;
            const res = await fetch(url, {
                method: 'POST',
                body: formData
            });

            return res.ok;
        }

        function downloadCSVFile(csvContent, filename) {
            const blob = new Blob([csvContent], { type: 'text/csv;charset=utf-8;' });
            const url = URL.createObjectURL(blob);
            const link = document.createElement('a');

            link.href = url;
            link.setAttribute('download', filename);
            document.body.appendChild(link);
            link.click();
            document.body.removeChild(link);

            setTimeout(() => URL.revokeObjectURL(url), 1000);
            return blob;
        }

        async function exportOrdersCSV(sendToBot = false) {
            if (!isAdminUser()) {
                return showAlert("Доступ разрешён только администратору");
            }

            const orders = getAllOrders();
            if (!orders.length) {
                return showAlert("Нет данных для выгрузки");
            }

            const csv = buildOrdersCSV();
            const filename = `max_orders_${new Date().toISOString().slice(0, 19).replace(/[:T]/g, '-')}.csv`;
            const blob = new Blob([csv], { type: 'text/csv;charset=utf-8;' });

            if (sendToBot) {
                try {
                    const ok = await sendCSVToAdmin(blob, filename);
                    if (ok) {
                        showAlert("CSV успешно отправлен администратору в бот");
                    } else {
                        showAlert("Не удалось отправить CSV в бот");
                    }
                } catch (e) {
                    showAlert("Ошибка отправки CSV");
                }
                return;
            }

            downloadCSVFile(csv, filename);
        }

        async function finalizeOrder() {
            const phone = document.getElementById('f-phone').value;
            const address = document.getElementById('f-address').value;

            if (!phone || !address) {
                return showAlert("Укажите телефон и адрес для связи");
            }

            if (!cart.length) {
                return showAlert("Корзина пуста");
            }

            const order = buildOrderObject(phone, address);

            try {
                // 1. отправляем сообщение о заказе
                const orderSent = await sendMessageToAdmin(order);
                if (!orderSent) {
                    return showAlert("Не удалось отправить заказ администратору");
                }

                // 2. сохраняем заказ локально
                saveOrderLocally(order);
                renderOrderHistory();

                // 3. формируем и отправляем CSV админу
                const csv = buildOrdersCSV();
                const filename = `max_orders_${new Date().toISOString().slice(0, 19).replace(/[:T]/g, '-')}.csv`;
                const blob = new Blob([csv], { type: 'text/csv;charset=utf-8;' });

                const csvSent = await sendCSVToAdmin(blob, filename);
                if (!csvSent) {
                    return showAlert("Заказ отправлен, но CSV отправить не удалось");
                }

                // 4. успех
                showConfirm("Заказ и CSV успешно отправлены администратору MAX. Очистить корзину?", (ok) => {
                    if (ok) {
                        cart = [];
                        localStorage.setItem(STORAGE_CART_KEY, JSON.stringify(cart));
                        document.getElementById('f-phone').value = phone;
                        document.getElementById('f-address').value = address;
                        updateCartUI();
                        switchTab('shop-view', document.querySelector('.nav-item'));
                    }
                });
            } catch (e) {
                showAlert("Ошибка связи с сервером MAX");
            }
        }

        function switchTab(id, el) {
            document.querySelectorAll('.view').forEach(v => v.classList.remove('active'));
            document.querySelectorAll('.nav-item').forEach(n => n.classList.remove('active'));

            const target = document.getElementById(id);
            if (target) target.classList.add('active');
            if (el) el.classList.add('active');

            if (id === 'profile-view') {
                renderProfile();
                renderOrderHistory();
            }

            safeSelection();
        }

        function isAdminUser() {

                if (lastOrder.address) document.getElementById('f-address').value = lastOrder.address;
            }
        }
